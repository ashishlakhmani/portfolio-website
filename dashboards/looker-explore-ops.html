-- Generated by Looker · bigquery_prod · Explore: production SELECT DATE(production.production_date) AS `production.date`, production.plant_name AS `production.plant`, shifts.shift_name AS `shifts.shift_name`, SUM(production.volume_tonnes) AS `production.total_volume_tonnes`, AVG(production.efficiency_pct) AS `production.avg_efficiency`, AVG(shifts.efficiency_pct) AS `shifts.avg_efficiency` FROM `ops_prod.production` AS production LEFT JOIN `ops_prod.shifts` AS shifts ON production.shift_id = shifts.shift_id WHERE production.production_date >= DATE_SUB(CURRENT_DATE(), INTERVAL 30 DAY) GROUP BY 1, 2, 3 ORDER BY 1 DESC, 4 DESC LIMIT 500
